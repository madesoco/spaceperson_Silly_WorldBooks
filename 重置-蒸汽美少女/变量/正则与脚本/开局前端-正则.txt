        /* 动态组合种族与职业技能 */
        function updateSkills() {
            var race = document.getElementById('v-race').value;
            var pclass = document.getElementById('v-class').value;

            var rSkill = raceSkills[race] || '';
            var cSkill = classSkills[pclass] || '';

            /* 更新UI提示字 */
            document.getElementById('hint-race').innerText = rSkill ? '获得种族特性：已同步' : '无特殊种族特性';
            document.getElementById('hint-class').innerText = cSkill ? '获得职业技能：已同步' : '无默认职业技能';

            var combined = [];
            if(rSkill) combined.push(rSkill);
            if(cSkill) combined.push(cSkill);

            document.getElementById('v-skills').value = combined.join('\n\n');
        }

        function handleStoryChange() {
            var val = document.getElementById('v-story').value;
            document.getElementById('story-desc').innerHTML = storyTemplates[val] || '';

            var bgLabel = document.getElementById('label-bg');
            if (val === '完全自定义开局') {
                bgLabel.innerHTML = '自定义设定 / 背景故事 <span style="color:#cd5c5c">* (自定义开局必填)</span>';
                if(document.getElementById('v-bg').value === '无') {
                    document.getElementById('v-bg').value = '';
                }
            } else {
                bgLabel.innerHTML = '自定义设定 / 背景故事';
                if(document.getElementById('v-bg').value.trim() === '') {
                    document.getElementById('v-bg').value = '无';
                }
            }
            checkForm();
        }

        function setDiff(points, btnEl) {
            var btns = document.getElementsByClassName('diff-btn');
            for(var i=0; i<btns.length; i++) {
                btns[i].classList.remove('active');
            }
            btnEl.classList.add('active');
            maxPoints = points;
            resetAttrs();
        }

        function resetAttrs() {
            attrs = { str: 8, dex: 8, con: 8, int: 8, wis: 8, cha: 8 };
            updateAttrDisplay();
        }

        function getUsedPoints() {
            return attrs.str + attrs.dex + attrs.con + attrs.int + attrs.wis + attrs.cha;
        }

        function modAttr(key, delta) {
            var used = getUsedPoints();
            if (delta > 0 && used >= maxPoints) return; /* 点数用完 */
            if (attrs[key] + delta < 1) return; /* 最小为1 */
            if (attrs[key] + delta > 20) return; /* 最大为20 */

            attrs[key] += delta;
            updateAttrDisplay();
        }

        function updateAttrDisplay() {
            document.getElementById('v-str').innerText = attrs.str;
            document.getElementById('v-dex').innerText = attrs.dex;
            document.getElementById('v-con').innerText = attrs.con;
            document.getElementById('v-int').innerText = attrs.int;
            document.getElementById('v-wis').innerText = attrs.wis;
            document.getElementById('v-cha').innerText = attrs.cha;

            var left = maxPoints - getUsedPoints();
            var el = document.getElementById('points-left');
            el.innerText = left;
            el.style.color = left === 0 ? '#cd5c5c' : '#d4af37';

            var btns = document.getElementsByClassName('attr-btn');
            for(var i=0; i<btns.length; i++) {
                if(btns[i].innerText === '+') {
                    btns[i].disabled = (left <= 0);
                }
            }
        }

        function checkForm() {
            var name = document.getElementById('v-name').value.trim();
            var story = document.getElementById('v-story').value;
            var bg = document.getElementById('v-bg').value.trim();

            var race = document.getElementById('v-race').value;
            var customRace = document.getElementById('v-custom-race').value.trim();

            var pclass = document.getElementById('v-class').value;
            var customClass = document.getElementById('v-custom-class').value.trim();

            var faith = document.getElementById('v-faith').value;
            var customFaith = document.getElementById('v-custom-faith').value.trim();

            var isValid = true;
            if (!name) isValid = false;
            if (story === '完全自定义开局' && (!bg || bg === '无')) isValid = false;
            if (race === '自定义' && !customRace) isValid = false;
            if (pclass === '自定义' && !customClass) isValid = false;
            if (faith === '自定义' && !customFaith) isValid = false;

            document.getElementById('btn-submit').disabled = !isValid;
        }

        function sanitizeText(str) {
            /* 过滤掉影响SillyTavern命令解析的竖线，并转为安全符号 */
            if(!str) return '无';
            return str.replace(/\|/g, '｜');
        }

        function formatYamlMultiLine(text) {
            /* 处理多行文本，保持YAML缩进 */
            var lines = text.split('\n');
            var result = '';
            for(var i=0; i<lines.length; i++) {
                if(lines[i].trim()) {
                    result += '\n      - ' + sanitizeText(lines[i].trim());
                }
            }
            return result || ' 无';
        }
