  function populateCharacterData() {
    const all_variables = getVariables({ type: 'message', message_id: getCurrentMessageId() });
    const stat = _.get(all_variables, 'stat_data', {});

    /* 全局信息 */
    $('#val-time').text(_.get(stat, '全局信息.时间', '未知'));
    $('#val-crown').text(_.get(stat, '全局信息.克朗', 0));
    $('#val-location').text(_.get(stat, '全局信息.地点', '未知'));

    /* 面板1: 主角信息 */
    $('#val-name').text(_.get(stat, '主角信息.姓名', '未知'));
    $('#val-gender').text(_.get(stat, '主角信息.性别', '未知'));
    $('#val-age').text(_.get(stat, '主角信息.年龄', '未知'));
    $('#val-race').text(_.get(stat, '主角信息.种族', '未知'));
    $('#val-identity').text(_.get(stat, '主角信息.身份', '无'));
    $('#val-faith').text(_.get(stat, '主角信息.信仰', '无'));
    $('#val-class').text(_.get(stat, '主角信息.职业', '无'));
    $('#val-adv-rank').text(_.get(stat, '主角信息.冒险者等级', '无'));
    $('#val-bg-sup').text(_.get(stat, '主角信息.背景补充', '无'));

    /* 面板2: 属性与技能 */
    const attrs = ['力量', '敏捷', '体质', '智力', '感知', '魅力'];
    attrs.forEach(attr => {
      $(`#val-attr-${attr}-num`).text(_.get(stat, `主角信息.属性.${attr}.数值`, 0));
      $(`#val-attr-${attr}-rank`).text(_.get(stat, `主角信息.属性.${attr}.位阶`, '未知'));
      $(`#val-attr-${attr}-desc`).html(_.get(stat, `主角信息.属性.${attr}.评价`, '暂无评价'));
    });

    const skills = _.get(stat, '主角信息.技能与能力', {});
    const skillsContainer = $('#skills-list');
    skillsContainer.empty();
    if (_.isEmpty(skills)) {
        skillsContainer.html('<div class="stb-vic-placeholder">暂无技能记录</div>');
    } else {
        let html = '';
        for (const [name, data] of Object.entries(skills)) {
            const desc = _.get(data, '描述', '无描述');
            const evalText = _.get(data, '评价', '');

            let contentHtml = desc;
            if (evalText) {
                contentHtml += `<hr class="stb-vic-divider"/><span class="stb-vic-eval-text">“${evalText}”</span>`;
            }

            html += `
            <details>
                <summary>
                    <span>${name}</span>
                    <span class="summary-extra">${_.get(data, '品质', '')}</span>
                    <button class="delete-btn" data-name="${name}" data-type="skill">删除</button>
                </summary>
                <div class="details-content">${contentHtml}</div>
            </details>`;
        }
        skillsContainer.html(html);
    }

    /* 面板3: 装备与背包 */
    const equips = _.get(stat, '主角信息.装备', {});
    const equipsContainer = $('#equips-list');
    equipsContainer.empty();
    if (_.isEmpty(equips)) {
        equipsContainer.html('<div class="stb-vic-placeholder">未装备任何物品</div>');
    } else {
        let html = '';
        for (const [name, data] of Object.entries(equips)) {
            html += `
            <details>
                <summary>
                    <span>${name}</span>
                    <span class="summary-extra">${_.get(data, '品质', '')}</span>
                    <button class="delete-btn" data-name="${name}" data-type="equip">删除</button>
                </summary>
                <div class="details-content">${_.get(data, '描述', '无描述')}</div>
            </details>`;
        }
        equipsContainer.html(html);
    }

    const bags = _.get(stat, '主角信息.背包', {});
    const bagsContainer = $('#bags-list');
    bagsContainer.empty();
    if (_.isEmpty(bags)) {
        bagsContainer.html('<div class="stb-vic-placeholder">背包空空如也</div>');
    } else {
        let html = '';
        for (const [name, data] of Object.entries(bags)) {
            html += `
            <details>
                <summary>
                    <span>${name} (${_.get(data, '数量', 1)})</span>
                    <span class="summary-extra">${_.get(data, '品质', '')}</span>
                    <button class="delete-btn" data-name="${name}" data-type="bag">删除</button>
                </summary>
                <div class="details-content">${_.get(data, '描述', '无描述')}</div>
            </details>`;
        }
        bagsContainer.html(html);
    }

    /* 面板4: 任务列表 */
    const tasks = _.get(stat, '任务列表.正在进行的任务', {});
    const tasksContainer = $('#tasks-list');
    tasksContainer.empty();
    if (_.isEmpty(tasks)) {
        tasksContainer.html('<div class="stb-vic-placeholder">当前没有任务</div>');
    } else {
        let html = '';
        for (const [name, data] of Object.entries(tasks)) {
            html += `
            <details>
                <summary>
                    <span>${_.get(data, '等级', '特殊')} · ${name}</span>
                    <span class="summary-extra">委托方: ${_.get(data, '委托方', '未知')}</span>
                    <button class="delete-btn" data-name="${name}" data-type="task">删除</button>
                </summary>
                <div class="details-content">
                    <p><span>详细：</span>${_.get(data, '详细', '无')}</p>
                    <p><span>目标：</span>${_.get(data, '目标', '无')}</p>
                    <p><span>奖励：</span>${_.get(data, '奖励', '无')}</p>
                </div>
            </details>`;
        }
        tasksContainer.html(html);
    }

    /* 面板5: 人物列表 */
    const npcs = _.get(stat, '人物列表', {});
    const npcsContainer = $('#npcs-list');
    npcsContainer.empty();
    if (_.isEmpty(npcs)) {
        npcsContainer.html('<div class="stb-vic-placeholder">暂未遇到重要人物</div>');
    } else {
        let html = '';
        for (const [name, data] of Object.entries(npcs)) {
            const inScene = _.get(data, '是否在场', false) ? ' [在场]' : '';
            html += `
            <details>
                <summary>
                    <span>${name} · ${_.get(data, '与玩家的关系', '未知')}${inScene}</span>
                    <span class="summary-extra">好感: ${_.get(data, '好感度', 0)}</span>
                    <button class="delete-btn" data-name="${name}" data-type="npc">删除</button>
                </summary>
                <div class="details-content">
                    <p><span>性别:</span> ${_.get(data, '性别', '未知')} | <span>年龄:</span> ${_.get(data, '年龄', '未知')} | <span>种族:</span> ${_.get(data, '种族', '未知')} | <span>等级:</span> ${_.get(data, '等级', '未知')}</p>
                    <p><span>身份:</span> ${_.get(data, '身份', '未知')}</p>
                    <p><span>外貌:</span> ${_.get(data, '外貌特征', '无')}</p>
                    <p><span>着装:</span> ${_.get(data, '着装', '无')}</p>
                    <p><span>性格:</span> ${_.get(data, '性格概括', '无')}</p>
                    <p><span>背景:</span> ${_.get(data, '背景故事', '无')}</p>
                    <p><span>心里话:</span> <span class="stb-vic-eval-text">“${_.get(data, '心里话', '...')}”</span></p>
                </div>
            </details>`;
        }
        npcsContainer.html(html);
    }
  }

  async function init() {
    await waitGlobalInitialized('Mvu');
    populateCharacterData();

    eventOn(Mvu.events.VARIABLE_UPDATE_ENDED, () => {
      populateCharacterData();
    });

    // 添加删除按钮事件监听器
    $(document).on('click', '.delete-btn', function() {
      const name = $(this).data('name');
      const type = $(this).data('type');
      let itemType = '项目';
      let path = '';

      switch(type) {
        case 'skill':
          itemType = '技能';
          path = '主角信息.技能与能力';
          break;
        case 'equip':
          itemType = '装备';
          path = '主角信息.装备';
          break;
        case 'bag':
          itemType = '背包物品';
          path = '主角信息.背包';
          break;
        case 'task':
          itemType = '任务';
          path = '任务列表.正在进行的任务';
          break;
        case 'npc':
        default:
          itemType = '人物';
          path = '人物列表';
          break;
      }

      if (confirm(`确定要删除${itemType} "${name}" 的所有信息吗？此操作不可撤销。`)) {
        const all_variables = getVariables({ type: 'message', message_id: getCurrentMessageId() });
        const stat = _.get(all_variables, 'stat_data', {});
        const targetObj = _.get(stat, path, {});
        if (targetObj[name]) {
          delete targetObj[name];
          _.set(stat, path, targetObj);
          replaceVariables(all_variables, { type: 'message', message_id: getCurrentMessageId() });
          // 重新填充数据
          populateCharacterData();
        }
      }
    });
  }

  $(errorCatched(init));
</script>
