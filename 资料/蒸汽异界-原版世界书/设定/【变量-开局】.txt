    function showTooltip(element, text) {
        clearTimeout(tooltipTimeout);
        const tooltip = DOM.locationTooltip;

        tooltip.textContent = text;
        tooltip.classList.add('visible');
        currentTooltipTarget = element;

        const updatePosition = () => {
            if (!element || !currentTooltipTarget) return;

            const rect = element.getBoundingClientRect();
            const tooltipRect = tooltip.getBoundingClientRect();
            const viewportHeight = window.innerHeight;
            const viewportWidth = window.innerWidth;

            let top = rect.bottom + 10;
            let left = rect.left + (rect.width / 2) - (tooltipRect.width / 2);

            tooltip.classList.remove('top');

            if (top + tooltipRect.height + 10 > viewportHeight) {
                top = rect.top - tooltipRect.height - 10;
                tooltip.classList.add('top');
            }

            if (left < 10) {
                left = 10;
            } else if (left + tooltipRect.width > viewportWidth - 10) {
                left = viewportWidth - tooltipRect.width - 10;
            }

            tooltip.style.top = `${top}px`;
            tooltip.style.left = `${left}px`;
        };

        requestAnimationFrame(updatePosition);
    }

    function hideTooltip() {

    }

            if (plusBtn) plusBtn.disabled = val >= 18 || appState.remainingPoints < costToIncrease;
            if (minusBtn) minusBtn.disabled = val <= 7;
        });

    function changeAttribute(attr, delta) {
        const oldValue = appState.attributes[attr];
        const newValue = oldValue + delta;
        if (newValue < 7 || newValue > 18) return;

        const oldCost = pointCost[oldValue];
        const newCost = pointCost[newValue];
        const costChange = newCost - oldCost;

        if (appState.remainingPoints - costChange >= 0) {
            appState.attributes[attr] = newValue;
            document.getElementById(`value-${attr}`).textContent = newValue;
            updateUI();
        }
